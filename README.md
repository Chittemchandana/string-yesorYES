# string-yesorYES
# problem1
# approach1(upper)
t=int(input())
for _ in range(t):
  s=input()
  s=s.upper()
  if s=="YES":
    print("YES")
  else:
    print("NO")

# approach2(lower)
t=int(input())
for _ in range(t):
  s=input()
  s=s.lower()
  if s=="yes":
    print("Yes")
  else:
    print("NO")
